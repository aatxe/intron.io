struct Point2D {
  x: i32,
  y: i32,
}

struct Point3D extends Point2D with {
  z: i32,
}

let origin2d = Point2D { x: 0, y: 0 }
let origin3d = Point3D { z: 0 | origin2d }

def projX(point: { x: i32 | a }): i32 = point.x

let main = {
  println(projX(origin2d));
  println(projX(origin3d));
}
